```python
"""PDF 加载与智能切分"""

import PyPDF2
from typing import List
import tiktoken

def load_pdf(file_path: str) -> str:
    """读取 PDF 并返回全文"""
    text = ""
    with open(file_path, "rb") as f:
        reader = PyPDF2.PdfReader(f)
        for page in reader.pages:
            text += page.extract_text() + "\n"
    return text.strip()

def split_text(
    text: str,
    max_tokens: int = 1000,
    overlap: int = 200,
    model: str = "gpt-4o-mini"
) -> List[str]:
    """按 token 切分文本，带重叠窗口"""
    enc = tiktoken.encoding_for_model(model)
    tokens = enc.encode(text)

    chunks = []
    start = 0
    while start < len(tokens):
        end = start + max_tokens
        chunk_tokens = tokens[start:end]
        chunks.append(enc.decode(chunk_tokens))
        start += max_tokens - overlap  # 重叠窗口

    return chunks

def estimate_tokens(text: str, model: str = "gpt-4o-mini") -> int:
    """估算 token 数"""
    enc = tiktoken.encoding_for_model(model)
    return len(enc.encode(text))
```

#### `src/llm_chain.py`

```python
"""LLM 调用封装（OpenAI SDK v1.0+）"""

import os
from openai import OpenAI
from typing import List, Dict, Optional

class LLMChain:
    def __init__(self, api_key: Optional[str] = None, base_url: Optional[str] = None):
        self.client = OpenAI(
            api_key=api_key or os.getenv("OPENAI_API_KEY"),
            base_url=base_url or os.getenv("OPENAI_BASE_URL")
        )
        self.model = os.getenv("MODEL_NAME", "gpt-4o-mini")
        self.max_tokens = int(os.getenv("MAX_TOKENS", "2000"))
        self.temperature = float(os.getenv("TEMPERATURE", "0.7"))

    def chat(
        self,
        system_msg: str,
        user_msg: str,
        chat_history: Optional[List[Dict]] = None
    ) -> str:
        """单次对话调用"""
        messages = [{"role": "system", "content": system_msg}]

        if chat_history:
            messages.extend(chat_history)

        messages.append({"role": "user", "content": user_msg})

        resp = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            max_tokens=self.max_tokens,
            temperature=self.temperature
        )
        return resp.choices[0].message.content

    def chat_stream(
        self,
        system_msg: str,
        user_msg: str
    ):
        """流式调用（Streamlit 用）"""
        messages = [
            {"role": "system", "content": system_msg},
            {"role": "user", "content": user_msg}
        ]

        resp = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            max_tokens=self.max_tokens,
            temperature=self.temperature,
            stream=True
        )

        for chunk in resp:
            if chunk.choices and chunk.choices[0].delta.content:
                yield chunk.choices[0].delta.content
```

```python
"""安全计算器工具"""

import operator
from typing import Dict, Any

ALLOWED_OPS = {
    "+": operator.add,
    "-": operator.sub,
    "*": operator.mul,
    "/": operator.truediv,
}

def calculate(expression: str) -> Dict[str, Any]:
    """
    安全计算表达式（仅支持加减乘除）
    警告：不要用 eval()！
    """
    try:
        # 解析简单表达式（如 "123 + 456"）
        parts = expression.strip().split()
        if len(parts) != 3:
            return {"error": "表达式格式错误，应为：数字 运算符 数字"}

        a, op, b = parts
        a, b = float(a), float(b)

        if op not in ALLOWED_OPS:
            return {"error": f"不支持运算符：{op}"}

        result = ALLOWED_OPS[op](a, b)
        return {"expression": expression, "result": result}
    except ZeroDivisionError:
        return {"error": "除数不能为 0"}
    except ValueError:
        return {"error": "数字格式错误"}
```

#### `agent.py`（ReAct 手写版）

```python
"""手写 ReAct Agent 循环"""

import json
import re
from openai import OpenAI
from tools.weather import get_weather, format_weather
from tools.calculator import calculate

client = OpenAI()

# 工具注册表
TOOLS = {
    "get_weather": get_weather,
    "calculate": calculate,
}

def run_agent(user_input: str, max_steps: int = 5):
    """
    ReAct 循环：Thought → Action → Observation → ...
    """
    messages = [
        {"role": "system", "content": """你是一个助手，可以使用工具回答问题。

可用工具：
- get_weather(city: str) -> 查询城市天气
- calculate(expression: str) -> 计算数学表达式

输出格式（严格遵守）：
Thought: 你的思考过程
Action: 工具名(参数)
"""},
        {"role": "user", "content": user_input}
    ]

    for step in range(max_steps):
        resp = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=messages,
            temperature=0
        )
        content = resp.choices[0].message.content
        print(f"Step {step + 1}:\n{content}\n{'-' * 50}")

        # 解析 Action
        action_match = re.search(r"Action:\s*(\w+)\((.+)\)", content)
        if not action_match:
            # 无 Action → 最终回答
            return content

        tool_name = action_match.group(1)
        tool_arg = action_match.group(2).strip("\"'")

        # 执行工具
        if tool_name in TOOLS:
            obs = TOOLS[tool_name](tool_arg)
            obs_str = json.dumps(obs, ensure_ascii=False)
        else:
            obs_str = f"错误：工具 {tool_name} 不存在"

        # 追加到消息
        messages.append({"role": "assistant", "content": content})
        messages.append({"role": "user", "content": f"Observation: {obs_str}\n\n继续思考..."})

    return "达到最大步数，未能完成回答"

if __name__ == "__main__":
    print(run_agent("深圳明天天气怎么样？帮我算一下 25 度加 3 度等于多少"))
```

```python
"""离线索引脚本：切分 → Embedding → 存入 Chroma"""

import os
from dotenv import load_dotenv
from llama_index.core import Document, VectorStoreIndex, StorageContext, load_index_from_storage
from llama_index.vector_stores.chroma import ChromaVectorStore
from llama_index.embeddings.openai import OpenAIEmbedding
import chromadb

load_dotenv()

def ingest_documents(data_dir: str, collection_name: str = "company_docs"):
    """从目录加载文档并建立索引"""
    from llama_index.core import SimpleDirectoryReader

    # 1. 加载文档
    reader = SimpleDirectoryReader(input_dir=data_dir)
    documents = reader.load_data()
    print(f"✅ 加载 {len(documents)} 个文档")

    # 2. 初始化 Chroma
    chroma_client = chromadb.PersistentClient(path="./chroma_db")
    chroma_collection = chroma_client.get_or_create_collection(collection_name)

    # 3. 构建索引
    embed_model = OpenAIEmbedding(
        model="text-embedding-3-small",
        api_key=os.getenv("OPENAI_API_KEY")
    )

    vector_store = ChromaVectorStore(chroma_collection=chroma_collection)
    storage_context = StorageContext.from_defaults(vector_store=vector_store)

    index = VectorStoreIndex.from_documents(
        documents,
        storage_context=storage_context,
        embed_model=embed_model
    )

    # 4. 持久化
    index.storage_context.persist()
    print(f"✅ 索引已保存到 chroma_db/")

if __name__ == "__main__":
    ingest_documents("data/raw")
```

#### `src/rag_chain.py`

```python
"""RAG 检索 + 生成"""

from llama_index.core import load_index_from_storage, StorageContext
from llama_index.llms.openai import OpenAI
from llama_index.embeddings.openai import OpenAIEmbedding
from llama_index.core.query_engine import RetrieverQueryEngine
from llama_index.core.retrievers import VectorIndexRetriever
from llama_index.core.postprocessor import SimilarityPostprocessor
import os

class RAGChain:
    def __init__(self, chroma_db_dir: str = "./chroma_db"):
        # 加载索引
        embed_model = OpenAIEmbedding(
            model="text-embedding-3-small",
            api_key=os.getenv("OPENAI_API_KEY")
        )
        storage_context = StorageContext.from_defaults(
            persist_dir=chroma_db_dir
        )
        self.index = load_index_from_storage(
            storage_context,
            embed_model=embed_model
        )

        # 配置 LLM
        self.llm = OpenAI(
            model=os.getenv("MODEL_NAME", "gpt-4o-mini"),
            temperature=0.1
        )

    def query(self, question: str, top_k: int = 3) -> dict:
        """执行 RAG 查询"""
        # 1. 检索
        retriever = VectorIndexRetriever(
            index=self.index,
            similarity_top_k=top_k
        )

        # 2. 后处理（相似度过滤）
        postprocessor = SimilarityPostprocessor(similarity_cutoff=0.7)

        # 3. 构建 Query Engine
        query_engine = RetrieverQueryEngine(
            retriever=retriever,
            node_postprocessors=[postprocessor],
            llm=self.llm
        )

        # 4. 执行查询
        response = query_engine.query(question)

        return {
            "answer": str(response),
            "source_nodes": [
                {
                    "text": node.node.text[:200],
                    "score": node.score,
                    "file": node.node.metadata.get("file_name", "未知")
                }
                for node in response.source_nodes
            ]
        }
```

```python
"""有记忆的对话 Agent（Redis 版）"""

import redis
import json
from openai import OpenAI
import os

client = OpenAI()

# Redis 连接
r = redis.Redis(host="localhost", port=6379, db=0, decode_responses=True)

def get_chat_history(session_id: str, max_turns: int = 5) -> list:
    """从 Redis 获取对话历史"""
    key = f"chat:{session_id}"
    # 保留最近 N 轮
    raw = r.lrange(key, -max_turns * 2, -1)
    return [json.loads(msg) for msg in raw]

def save_message(session_id: str, role: str, content: str):
    """保存消息到 Redis"""
    key = f"chat:{session_id}"
    r.rpush(key, json.dumps({"role": role, "content": content}))
    r.expire(key, 60 * 60 * 24 * 30)  # 30 天过期

def chat_with_memory(user_input: str, session_id: str = "default") -> str:
    """带记忆的对话"""
    # 1. 获取历史
    history = get_chat_history(session_id)

    # 2. 构建消息
    messages = [
        {"role": "system", "content": "你是一个有记忆的助手，能记住之前的对话。"}
    ]
    messages.extend(history)
    messages.append({"role": "user", "content": user_input})

    # 3. 调用 LLM
    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=messages
    )
    answer = resp.choices[0].message.content

    # 4. 保存对话
    save_message(session_id, "user", user_input)
    save_message(session_id, "assistant", answer)

    return answer

if __name__ == "__main__":
    print("开始对话（输入 'exit' 退出）")
    session = "user_001"
    while True:
        user_input = input("你：")
        if user_input == "exit":
            break
        answer = chat_with_memory(user_input, session)
        print(f"AI：{answer}")
```

```python
"""E2B 代码沙箱封装"""

from e2b import Sandbox
from typing import Dict
import os

class CodeSandbox:
    def __init__(self):
        self.sandbox = Sandbox(
            api_key=os.getenv("E2B_API_KEY"),
            timeout=300  # 5 分钟超时
        )

    def run_python(self, code: str, files: Dict[str, bytes] = None) -> Dict:
        """
        在沙箱中执行 Python 代码
        Returns:
            {"stdout": ..., "stderr": ..., "exit_code": ...}
        """
        # 1. 上传文件（如有）
        if files:
            for name, content in files.items():
                self.sandbox.files.write(f"/home/user/{name}", content)

        # 2. 执行代码
        execution = self.sandbox.process.start_and_wait(
            cmd=["python3", "-c", code],
            timeout=60
        )

        return {
            "stdout": execution.stdout,
            "stderr": execution.stderr,
            "exit_code": execution.exit_code
        }

    def close(self):
        self.sandbox.close()
```

#### `workflow.py`

```python
"""LangGraph 多 Agent 工作流"""

from langgraph.graph import StateGraph, END
from state import ArticleState
from agents.researcher import research_node
from agents.writer import write_node
from agents.reviewer import review_node

def should_revise(state: ArticleState) -> str:
    """条件边：决定是否需要修改"""
    if state.get("feedback") and state["iteration_count"] < 3:
        return "revise"
    return "publish"

def build_workflow():
    """构建工作流"""
    workflow = StateGraph(ArticleState)

    # 1. 添加节点
    workflow.add_node("researcher", research_node)
    workflow.add_node("writer", write_node)
    workflow.add_node("reviewer", review_node)
    workflow.add_node("reviser", write_node)  # 复用 writer 做修改

    # 2. 添加边
    workflow.set_entry_point("researcher")
    workflow.add_edge("researcher", "writer")
    workflow.add_edge("writer", "reviewer")

    # 3. 条件边
    workflow.add_conditional_edges(
        "reviewer",
        should_revise,
        {
            "revise": "reviser",
            "publish": END
        }
    )
    workflow.add_edge("reviser", "reviewer")

    return workflow.compile()

if __name__ == "__main__":
    app = build_workflow()
    result = app.invoke({"topic": "2026 年 AI Agent 技术趋势"})
    print(result["final_article"])
```

```python
"""Token 成本追踪"""

import time
import json
from datetime import datetime
from typing import Dict, Optional

# 2026 年 6 月定价（美元 / 1M tokens）
PRICING = {
    "gpt-4o": {"input": 5.0, "output": 15.0},
    "gpt-4o-mini": {"input": 0.15, "output": 0.6},
    "gpt-4-turbo": {"input": 10.0, "output": 30.0},
    "claude-3.5-sonnet": {"input": 3.0, "output": 15.0},
    "claude-3.5-haiku": {"input": 1.0, "output": 5.0},
}

class CostTracker:
    def __init__(self, log_file: str = "cost_log.jsonl"):
        self.log_file = log_file
        self.session_cost = 0.0

    def calc_cost(self, model: str, input_tokens: int, output_tokens: int) -> float:
        """计算单次调用成本"""
        if model not in PRICING:
            return 0.0

        prices = PRICING[model]
        cost = (input_tokens / 1_000_000 * prices["input"] +
                output_tokens / 1_000_000 * prices["output"])
        return round(cost, 6)

    def log_call(
        self,
        model: str,
        input_tokens: int,
        output_tokens: int,
        user_id: Optional[str] = None,
        latency_ms: Optional[int] = None
    ):
        """记录一次 LLM 调用"""
        cost = self.calc_cost(model, input_tokens, output_tokens)
        self.session_cost += cost

        log_entry = {
            "timestamp": datetime.utcnow().isoformat(),
            "model": model,
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
            "cost_usd": cost,
            "user_id": user_id,
            "latency_ms": latency_ms
        }

        with open(self.log_file, "a") as f:
            f.write(json.dumps(log_entry, ensure_ascii=False) + "\n")

        # 成本预警
        if self.session_cost > 10.0:
            print(f"⚠️ 警告：当前会话成本 ${self.session_cost:.2f}")

    def get_daily_cost(self, date: Optional[str] = None) -> float:
        """统计某天的总成本"""
        total = 0.0
        with open(self.log_file, "r") as f:
            for line in f:
                entry = json.loads(line)
                if date is None or entry["timestamp"].startswith(date):
                    total += entry["cost_usd"]
        return round(total, 2)
```

```python
"""输入过滤：Prompt 注入检测"""

import re
from typing import Dict, Tuple

# 注入攻击模式（简化版）
INJECTION_PATTERNS = [
    r"忽略.{0,20}之前的指令",
    r"ignore.{0,20}previous.{0,20}instructions",
    r"忘记.{0,20}你的系统提示",
    r"forget.{0,20}system.{0,20}prompt",
    r"你现在是一个",
    r"you are now a",
    r"\[SYSTEM\]",
    r"### 系统：",
]

class InputFilter:
    def __init__(self, sensitivity: float = 0.7):
        self.sensitivity = sensitivity

    def detect_injection(self, user_input: str) -> Dict:
        """
        检测 Prompt 注入攻击
        Returns:
            {"is_attack": bool, "score": float, "matched_patterns": list}
        """
        matched = []
        for pattern in INJECTION_PATTERNS:
            if re.search(pattern, user_input, re.IGNORECASE):
                matched.append(pattern)

        score = len(matched) / len(INJECTION_PATTERNS)
        is_attack = score >= self.sensitivity

        return {
            "is_attack": is_attack,
            "score": round(score, 2),
            "matched_patterns": matched,
            "action": "block" if is_attack else "allow"
        }

    def sanitize(self, user_input: str) -> str:
        """简单脱敏（示例）"""
        # 移除可能的 API Key
        sanitized = re.sub(r"sk-[a-zA-Z0-9]{48}", "[API_KEY_REDACTED]", user_input)
        return sanitized
```

```python
"""GitHub Webhook 接收器"""

from fastapi import APIRouter, Request, HTTPException
from typing import Dict
import hmac
import hashlib
import os

router = APIRouter()

@router.post("/github/webhook")
async def github_webhook(request: Request):
    """接收 GitHub Webhook"""
    # 1. 验证签名
    body = await request.body()
    signature = request.headers.get("X-Hub-Signature-256")
    if not verify_signature(body, signature):
        raise HTTPException(status_code=403, detail="Invalid signature")

    # 2. 解析事件
    event_type = request.headers.get("X-GitHub-Event")
    payload = await request.json()

    if event_type == "pull_request" and payload["action"] in ["opened", "synchronize"]:
        # 触发 Code Review
        await trigger_review(payload)

    return {"status": "ok"}

def verify_signature(body: bytes, signature: str) -> bool:
    """验证 GitHub Webhook 签名"""
    secret = os.getenv("GITHUB_WEBHOOK_SECRET").encode()
    mac = hmac.new(secret, body, hashlib.sha256)
    expected = "sha256=" + mac.hexdigest()
    return hmac.compare_digest(expected, signature)
```
