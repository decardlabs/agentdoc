**路由策略（省钱大法）**：

```python
from dataclasses import dataclass
from typing import Literal

@dataclass
class RouterConfig:
    """多模型路由配置"""
    # 任务分类
    simple_tasks: list = ("greeting", "faq", "status_check")
    complex_tasks: list = ("code_review", "financial_analysis", "multi_step")

def route_model(task_type: str) -> str:
    """根据任务复杂度路由到不同模型"""
    if task_type in ("code_review", "financial_analysis"):
        return "gpt-4o"          # 复杂任务用大模型
    elif task_type in ("greeting", "faq"):
        return "gpt-4o-mini"     # 简单任务用小模型
    else:
        return "gpt-4o-mini"     # 默认小模型

# 使用缓存进一步省钱
from functools import lru_cache
import hashlib

class CachedLLM:
    def __init__(self):
        self.cache = {}

    def call_with_cache(self, prompt: str, model: str, ttl: int = 3600):
        cache_key = hashlib.md5(f"{model}:{prompt}".encode()).hexdigest()
        if cache_key in self.cache:
            hit = self.cache[cache_key]
            if time.time() - hit["ts"] < ttl:
                return hit["data"]
        # 没命中缓存，调 API
        result = self._call_api(prompt, model)
        self.cache[cache_key] = {"data": result, "ts": time.time()}
        return result
```

#### Agent 框架选型

| 框架 | 适用场景 | 优势 | 劣势 | 推荐 |
|------|----------|------|------|------|
| **LangChain + LangGraph** | 复杂 Agent、多步流程、需要状态管理 | 生态最大、教程最多、社区活跃 | 抽象层太多、调试困难、版本迭代快 | ★★★★★ 首选 |
| **LlamaIndex** | RAG 为核心的应用 | RAG 专项最优、索引管理强 | Agent 能力弱于 LangChain | ★★★★☆ RAG 场景 |
| **AutoGen (微软)** | 多 Agent 对话、代码生成 | 多 Agent 原生命支持、代码沙箱 | 学习曲线陡、文档零散 | ★★★☆☆ 多Agent |

| **手写 (纯 OpenAI SDK)** | 简单 Agent、教学目的 | 完全可控、无框架依赖 | 重复造轮子、状态管理靠自己 | ★★☆☆☆ 简单场景 |

**决策法则**：

```
你的 Agent 需要状态图吗？（if/else 分支、循环、人工审批）
  → 是：LangGraph
  → 否：继续判断

你的核心是检索吗？
  → 是：LlamaIndex
  → 否：继续判断

你有 3+ 个 Agent 需要互相通信吗？
  → 是：CrewAI 或 AutoGen
  → 否：LangChain Agent (最省事)

你是教学演示 / 只有 1-2 个工具吗？
  → 是：手写 OpenAI SDK（理解原理）
  → 否：选框架
```

    data: Optional[str] = None
    error: Optional[str] = None
    latency_ms: float = 0.0
    tokens_used: int = 0

class ToolRegistry:
    """生产级工具注册表"""

    def __init__(self):
        self._tools: Dict[str, Callable] = {}
        self._schemas: Dict[str, Dict] = {}
        self._stats: Dict[str, Dict] = {}  # 调用统计

    def register(self, func: Callable, schema: Dict):
        """注册工具"""
        name = schema["function"]["name"]
        self._tools[name] = func
        self._schemas[name] = schema
        self._stats[name] = {
            "total_calls": 0,
            "total_failures": 0,
            "total_latency_ms": 0
        }

    def execute(self, name: str, args: Dict) -> ToolResult:
        """执行工具，带统计和降级"""
        start = time.time()
        if name not in self._tools:
            return ToolResult(
                success=False,
                error=f"未知工具: {name}"
            )

        try:
            func = self._tools[name]
            result = func(**args)
            latency = (time.time() - start) * 1000

            # 更新统计
            self._stats[name]["total_calls"] += 1
            self._stats[name]["total_latency_ms"] += latency

            logger.info(
                f"Tool executed: {name} | "
                f"Args: {args} | "
                f"Latency: {latency:.0f}ms"
            )

            return ToolResult(
                success=True,
                data=str(result),
                latency_ms=latency
            )

        except Exception as e:
            latency = (time.time() - start) * 1000
            self._stats[name]["total_failures"] += 1

            logger.error(
                f"Tool failed: {name} | "
                f"Args: {args} | "
                f"Error: {str(e)}"
            )

            return ToolResult(
                success=False,
                error=f"执行失败: {str(e)}",
                latency_ms=latency
            )

    def get_all_schemas(self) -> List[Dict]:
        """获取所有工具的 OpenAI Function Schema"""
        return list(self._schemas.values())

    def get_stats(self) -> Dict:
        """获取工具调用统计"""
        return self._stats

import re
from typing import Tuple, Optional

class SecurityGuard:
    """安全防护层（三层）"""

    # 第一层：正则过滤
    SENSITIVE_PATTERNS = {
        "id_card": r'\b\d{17}[\dXx]\b',
        "phone": r'\b1[3-9]\d{9}\b',
        "bank_card": r'\b\d{16,19}\b',
        "api_key": r'(sk-[a-zA-Z0-9]{20,})|(AIza[0-9A-Za-z_-]{35})',
    }

    # 第二层：Prompt 注入特征
    INJECTION_PATTERNS = [
        r'忽略[以之]?[上前]的(所有)?指令',
        r'ignore (all )?previous instructions',
        r'forget (all )?previous',
        r'you are now',
        r'扮演',
        r'pretend you are',
        r'\[SYSTEM\]',
        r'<\|im_start\|>',
    ]

    # 第三层：内容审核（调用审核 API）
    BLOCKED_CATEGORIES = ["hate", "violence", "self-harm", "sexual"]

    def check_input(self, text: str) -> Tuple[bool, Optional[str]]:
        """
        检查用户输入
        返回: (是否通过, 拦截原因)
        """
        # 第一层：脱敏（不阻断，只打码）
        text = self._mask_sensitive(text)

        # 第二层：注入检测
        for pattern in self.INJECTION_PATTERNS:
            if re.search(pattern, text, re.IGNORECASE):
                return False, f"检测到潜在注入攻击: {pattern}"

        # 第三层：内容审核
        if not self._content_moderation(text):
            return False, "内容审核未通过"

        return True, None

    def _mask_sensitive(self, text: str) -> str:
        """脱敏处理"""
        for name, pattern in self.SENSITIVE_PATTERNS.items():
            text = re.sub(pattern, f"[{name.upper()}_HIDDEN]", text)
        return text

    def _content_moderation(self, text: str) -> bool:
        """调用审核 API"""
        # 示例：OpenAI Moderation API
        import openai
        response = openai.moderations.create(input=text)
        return not response.results[0].flagged

class MonitoredLLM:
    """带监控的 LLM 客户端"""

        self.model = model

        status = "success"
        try:
            yield
        except Exception as e:
            status = type(e).__name__
            raise
        finally:

                time.time() - start

            LLM_CALL_TOTAL.labels(
                model=self.model,
                status=status

    async def create(self, messages, **kwargs):
        with self._monitor():
            response = await self.client.chat.completions.create(
                model=self.model,

                model=self.model,
                type="input"
            ).inc(usage.prompt_tokens)
            LLM_TOKEN_USAGE.labels(
                model=self.model,
                type="output"
            ).inc(usage.completion_tokens)

            return response

1. 安全问题 (SQL注入、XSS、SSRF、权限绕过)
2. 逻辑错误 (空指针、类型错误、边界条件)
3. 性能问题 (N+1 查询、不必要的循环、内存泄漏)
4. 代码风格 (命名、注释、复杂度)

## 输出格式
返回 JSON 数组，每个元素：
{
  "file": "路径",
  "line": 行号,
  "severity": "WARN|ERROR|INFO",
  "category": "security|logic|performance|style",
  "message": "简短描述",
  "suggestion": "具体修改建议"
}

## 规则
- 只评论有问题的代码，没问题就返回空数组
- 不要评论 import 语句、配置文件
- suggestion 必须具体可执行，不要写「建议优化」
"""

    def __init__(self, llm_client, github_client):
        self.llm = llm_client
        self.github = github_client

    async def review_pr(self, pr_data: Dict) -> List[ReviewComment]:
        """审查 PR"""
        comments = []

        # Step 1: 规则引擎（快速检查）
        for file_change in pr_data.get("changes", []):
            rule_comments = self._run_security_rules(
                file_change["filename"],
                file_change["patch"]
            )
            comments.extend(rule_comments)

        # Step 2: LLM 深度审查（只审 Python/JS/TS/Go 文件）
        code_files = [
            f for f in pr_data.get("changes", [])
            if f["filename"].endswith(('.py', '.js', '.ts', '.go'))
        ]
        if code_files:
            llm_comments = await self._llm_review(code_files)
            comments.extend(llm_comments)

        # Step 3: 去重 + 排序
        comments = self._deduplicate_comments(comments)
        comments.sort(key=lambda c: (
            0 if c.severity == "ERROR" else 1 if c.severity == "WARN" else 2
        ))

        # Step 4: 发布评论到 GitHub
        if comments:
            await self._post_review_to_github(
                pr_data["number"],
                comments
            )

        return comments

    def _run_security_rules(
        self, filename: str, patch: str
    ) -> List[ReviewComment]:
        """运行安全规则"""
        comments = []
        for rule_name, pattern in self.SECURITY_RULES.items():
            import re
            matches = re.finditer(pattern, patch, re.IGNORECASE)
            for match in matches:
                line_num = self._get_line_number(patch, match.start())
                comments.append(ReviewComment(
                    file_path=filename,
                    line_start=line_num,
                    line_end=line_num,
                    severity="ERROR",
                    category="security",
                    message=f"安全风险: {rule_name}",
                    suggestion=self._get_suggestion_for_rule(rule_name)
                ))
        return comments

    async def _llm_review(
        self, files: List[Dict]
    ) -> List[ReviewComment]:
        """LLM 深度审查"""
        # 构建审查提示
        code_context = ""
        for f in files[:5]:  # 最多审 5 个文件，控制成本
            code_context += f"\n### {f['filename']}\n```\n{f.get('patch', '')[:3000]}\n```\n"

        if not code_context.strip():
            return []

        response = await self.llm.create(
            messages=[
                {"role": "system", "content": self.SYSTEM_PROMPT},
                {"role": "user", "content": code_context}
            ],
            response_format={"type": "json_object"},
            temperature=0.1
        )

        try:
            results = json.loads(response.choices[0].message.content)
            return [
                ReviewComment(
                    file_path=r["file"],
                    line_start=r["line"],
                    line_end=r["line"],
                    severity=r["severity"],
                    category=r["category"],
                    message=r["message"],
                    suggestion=r.get("suggestion")
                )
                for r in results.get("reviews", [])
            ]
        except json.JSONDecodeError:
            return []

    def _deduplicate_comments(
        self, comments: List[ReviewComment]
    ) -> List[ReviewComment]:
        """去重"""
        seen = set()
        result = []
        for c in comments:
            key = (c.file_path, c.line_start, c.category)
            if key not in seen:
                seen.add(key)
                result.append(c)
        return result

    async def _post_review_to_github(
        self, pr_number: int, comments: List[ReviewComment]
    ):
        """发布 Review 到 GitHub"""
        # 用 GitHub API 创建 Review
        review_body = self._format_review_body(comments)
        await self.github.create_review(
            pr_number=pr_number,
            body=review_body,
            event="COMMENT"
        )

    def _format_review_body(
        self, comments: List[ReviewComment]
    ) -> str:
        """格式化 Review 内容"""
        lines = [
            "## Automated Code Review by Agent",
            "",
            f"Found **{len(comments)}** issue(s):",
            "",
            "| Severity | Category | File | Line | Message |",
            "|----------|----------|------|------|---------|",
        ]

        for c in comments:
            lines.append(
                f"| {c.severity} | {c.category} | "
                f"`{c.file_path}` | L{c.line_start} | "
                f"{c.message} |"
            )

        lines.append("")
        lines.append("### Details")
        for c in comments:
            lines.append(f"- **[{c.severity}] {c.file_path}:{c.line_start}**")
            lines.append(f"  - {c.message}")
            if c.suggestion:
                lines.append(f"  - Suggestion: {c.suggestion}")

        return "\n".join(lines)

    def _get_line_number(self, patch: str, char_pos: int) -> int:
        """根据字符位置计算行号"""
        return patch[:char_pos].count('\n') + 1

    def _get_suggestion_for_rule(self, rule_name: str) -> str:
        """获取规则对应的修复建议"""
        suggestions = {
            "sql_injection": "使用参数化查询 (cursor.execute(sql, params))",
            "hardcoded_secret": "使用环境变量 os.getenv('SECRET') 或密钥管理服务",
            "eval_usage": "使用 ast.literal_eval() 替代 eval()",
            "os_system": "使用 subprocess.run() 替代 os.system()",
            "subprocess_shell": "移除 shell=True，使用列表参数",
            "debug_true": "生产环境设置 DEBUG=False",
        }
        return suggestions.get(rule_name, "请检查代码安全性")
```

# app/core/agent/intent_router.py

from enum import Enum
from typing import Dict, Any

class Intent(str, Enum):
    FAQ = "faq"                    # 常见问题 → RAG 检索
    ORDER_STATUS = "order_status"  # 订单查询 → 工单 API
    COMPLAINT = "complaint"        # 投诉 → 转人工
    REFUND = "refund"              # 退款 → 转人工 + 创建工单
    TECH_ISSUE = "tech_issue"      # 技术问题 → 知识库 + 创建工单
    UNKNOWN = "unknown"            # 未知 → 追问或转人工

INTENT_PROMPT = """分析用户消息，判断意图。返回 JSON。

意图列表：
- faq: 常见问题、操作指引、产品说明
- order_status: 查询订单/工单状态
- complaint: 投诉、不满、要求投诉
- refund: 要求退款、退费
- tech_issue: 技术故障、报错、Bug
- unknown: 无法判断

消息: {message}

返回格式: {{"intent": "faq", "confidence": 0.95, "reason": "用户在问如何使用功能"}}
"""

class IntentRouter:
    """意图路由"""

    ROUTING_RULES = {
        Intent.COMPLAINT: "escalate_to_human",   # 投诉直接转人工
        Intent.REFUND: "create_refund_ticket",    # 退款创建工单
        Intent.ORDER_STATUS: "check_order_status", # 查询调用 API
        Intent.FAQ: "search_knowledge_base",       # FAQ 检索知识库
        Intent.TECH_ISSUE: "search_knowledge_base",# 技术问题也检索
        Intent.UNKNOWN: "ask_clarification",       # 未知就追问
    }

    def __init__(self, llm_client):
        self.llm = llm_client

    async def classify(self, message: str) -> Dict:
        """识别用户意图"""
        # 先用关键词快速匹配（省钱）
        quick_match = self._keyword_match(message)
        if quick_match and quick_match["confidence"] > 0.9:
            return quick_match

        # 关键词不匹配，用 LLM
        response = await self.llm.create(
            messages=[
                {"role": "user", "content": INTENT_PROMPT.format(message=message)}
            ],
            response_format={"type": "json_object"},
            temperature=0.0
        )

        import json
        result = json.loads(response.choices[0].message.content)
        return result

    def _keyword_match(self, message: str) -> Dict | None:
        """关键词快速匹配"""
        keywords = {
            Intent.COMPLAINT: ["投诉", "举报", "我要投诉", "差评"],
            Intent.REFUND: ["退款", "退钱", "退费", "退货"],
            Intent.ORDER_STATUS: ["订单", "工单", "进度", "处理得怎么样"],
        }

        for intent, words in keywords.items():
            for word in words:
                if word in message:
                    return {
                        "intent": intent.value,
                        "confidence": 0.95,
                        "reason": f"关键词匹配: {word}"
                    }
        return None

    async def route(
        self, intent: str, message: str, session_id: str
    ) -> Dict:
        """根据意图分发到对应处理器"""
        action = self.ROUTING_RULES.get(Intent(intent))

        if action == "escalate_to_human":
            return await self._handle_escalation(message, session_id)

        elif action == "create_refund_ticket":
            return await self._handle_refund(message, session_id)

        elif action == "check_order_status":
            return await self._handle_order_status(message, session_id)

        elif action == "search_knowledge_base":
            return await self._handle_knowledge_search(message, session_id)

        else:
            return {
                "response": "抱歉我没有完全理解您的问题。您可以换一种方式描述，或者输入「转人工」联系在线客服。",
                "type": "clarification"
            }

    async def _handle_escalation(
        self, message: str, session_id: str
    ) -> Dict:
        """处理转人工"""
        return {
            "response": "已为您转接人工客服，预计等待 1-2 分钟。您可以先描述您的问题，客服接入后会立即看到。",
            "type": "escalation",
            "session_id": session_id,
            "status": "waiting_for_human"
        }

    async def _handle_refund(
        self, message: str, session_id: str
    ) -> Dict:
        """处理退款"""
        # 创建退款工单
        ticket_id = await self.create_ticket(
            type="refund",
            description=message,
            session_id=session_id
        )
        return {
            "response": f"已为您创建退款工单 (编号: {ticket_id})，客服将在 30 分钟内电话联系您确认。",
            "type": "refund_ticket",
            "ticket_id": ticket_id
        }

    async def _handle_knowledge_search(
        self, message: str, session_id: str
    ) -> Dict:
        """处理知识库搜索"""
        # RAG 检索
        results = await self.rag.query(message)

        if results["answer"]:
            sources_text = "、".join([
                f"「{s['file']}」" for s in results["sources"][:3]
            ])
            return {
                "response": f"{results['answer']}\n\n参考来源: {sources_text}",
                "type": "knowledge",
                "sources": results["sources"]
            }
        else:
            # 没找到，创建工单
            return {
                "response": "抱歉，知识库暂未收录这个问题。已为您创建工单，技术团队会跟进。",
                "type": "ticket_created"
            }
```

        for md_file in self.vault_path.rglob("*.md"):
            # 跳过 .obsidian 目录和模板目录
            if ".obsidian" in str(md_file) or "templates" in str(md_file):
                continue

            note = self._parse_note(md_file)
            if note:
                notes.append(note)

        return notes

    def _parse_note(self, file_path: Path) -> Optional[Dict]:
        """解析单个笔记"""
        with open(file_path, "r", encoding="utf-8") as f:
            content = f.read()

        if not content.strip():
            return None

        # 提取 Frontmatter
        metadata = {}
        body = content
        if content.startswith("---"):
            parts = content.split("---", 2)
            if len(parts) >= 3:
                try:
                    metadata = yaml.safe_load(parts[1]) or {}
                except yaml.YAMLError:
                    pass
                body = parts[2]

        # 提取 WikiLinks: [[xxx]] 或 [[xxx|alias]]
        wikilinks = re.findall(r'\[\[([^\]|#]+)(?:\|[^\]]+)?\]\]', body)

        # 提取 Tags: #tag 或 #nested/tag
        tags = re.findall(r'#([\w/-]+)', body)

        # 提取标题（第一个 # 开头的行）
        title_match = re.search(r'^#\s+(.+)$', body, re.MULTILINE)
        title = title_match.group(1) if title_match else file_path.stem

        # 计算文件哈希
        file_hash = hashlib.md5(content.encode()).hexdigest()

        return {
            "id": file_hash,
            "title": title,
            "path": str(file_path.relative_to(self.vault_path)),
            "abs_path": str(file_path),
            "tags": list(set(tags)),
            "wikilinks": list(set(wikilinks)),
            "metadata": metadata,
            "content": content,
            "body": body.strip(),
            "created": datetime.fromtimestamp(
                os.path.getctime(file_path)
            ).isoformat(),
            "modified": datetime.fromtimestamp(
                os.path.getmtime(file_path)
            ).isoformat(),
            "word_count": len(body.split()),
            "hash": file_hash
        }

    def get_changed_files(self) -> tuple[List[Dict], List[str]]:
        """增量更新：返回新增/修改的笔记和被删除的路径"""
        new_or_updated = []
        current_paths = set()
        deleted = []

        for md_file in self.vault_path.rglob("*.md"):
            if ".obsidian" in str(md_file) or "templates" in str(md_file):
                continue

            rel_path = str(md_file.relative_to(self.vault_path))
            current_paths.add(rel_path)

            note = self._parse_note(md_file)
            if not note:
                continue

            old_hash = self.file_hashes.get(rel_path)
            if note["hash"] != old_hash:
                new_or_updated.append(note)
                self.file_hashes[rel_path] = note["hash"]

        # 删除不存在的
        for old_path in list(self.file_hashes.keys()):
            if old_path not in current_paths:
                deleted.append(old_path)
                del self.file_hashes[old_path]

        return new_or_updated, deleted

class VaultSearchAgent:
    """Vault 搜索 Agent — 向量 + 图 混合检索"""

    def __init__(self, vector_store, graph_store):
        self.vector_store = vector_store   # Qdrant
        self.graph_store = graph_store     # 简单邻接表

    async def search(
        self,
        query: str,
        mode: str = "hybrid",
        top_k: int = 5
    ) -> Dict:
        """
        混合检索
        mode: "vector" | "graph" | "hybrid"
        """
        if mode == "vector":
            return await self._vector_search(query, top_k)
        elif mode == "graph":
            return await self._graph_search(query, top_k)
        else:  # hybrid
            vector_results = await self._vector_search(query, top_k * 2)
            graph_results = await self._graph_expand(
                vector_results["notes"], top_k
            )
            return {
                "query": query,
                "mode": "hybrid",
                "notes": graph_results,
                "explanation": f"向量检索找到 {len(vector_results['notes'])} 篇, 图谱扩展后返回 {len(graph_results)} 篇"
            }

    async def _vector_search(
        self, query: str, top_k: int
    ) -> Dict:
        """纯向量检索"""
        results = self.vector_store.search(
            collection_name="obsidian_notes",
            query_text=query,
            limit=top_k
        )
        return {
            "query": query,
            "mode": "vector",
            "notes": [
                {
                    "title": r.payload["title"],
                    "path": r.payload["path"],
                    "snippet": r.payload["body"][:200],
                    "score": r.score,
                    "tags": r.payload.get("tags", [])
                }
                for r in results
            ]
        }

    async def _graph_expand(
        self, seed_notes: List[Dict], max_results: int
    ) -> List[Dict]:
        """基于图谱扩展推荐"""
        seen = set()
        results = []

        for note in seed_notes:
            note_id = note["path"]
            if note_id not in seen:
                seen.add(note_id)
                results.append(note)

            # 找相关的笔记（通过 WikiLink 关系）
            related = self.graph_store.get_related(note_id)
            for rel_note in related:
                if rel_note["path"] not in seen and len(results) < max_results:
                    seen.add(rel_note["path"])
                    results.append({
                        **rel_note,
                        "relation": f"链接自「{note['title']}」"
                    })

            if len(results) >= max_results:
                break

        return results

## 9. 常见翻车点

| 翻车点 | 表现 | 怎么避免 |
|--------|------|----------|
| **选题太大** | 想做「万能 Agent」，2 周后发现只做了 10% | 第一周必须确定 MVP 范围，列出「不做」列表 |
| **追求完美** | 纠结选哪个框架、哪个模型，迟迟不写代码 | 默认选最熟悉的，第一时间跑通最小闭环 |
| **忽视演示** | Demo 当天服务器挂了/网断了/Token 用完了 | 提前 3 天彩排，准备录屏作为备份 |
| **文档缺失** | 代码写完了，README 只有 3 行「跑一下就知道」 | 每天写一点文档，不要最后 1 天赶 |
| **安全裸奔** | 没有认证、没有注入防护、没有审核 | 安全从第一天就做，不要「最后再加」 |
| **成本失控** | 不知道花了多少钱，月底发现 Token 账单 ¥500+ | 从第一天加成本追踪，设置日预算上限 |
| **单打独斗** | 遇到问题自己憋着，不找导师/同学 | 每天群里汇报进度，遇到卡点立即求助 |
| **不看评分标准** | 做了很多评分标准不要求的东西，漏了要打分的东西 | 把评分表打印出来贴在显示器旁边 |
